# Modal

The modal component provides the functionality to toggle dialogs that overlays
the main content.

## Usage

In the usage examples below, we use the [Portal](/components/portal) component
to render the modal into a different part of the DOM. This is not required,
but it's recommended to avoid styling issues.

<DemoProvider>
  <DemoModal />
</DemoProvider>

```tsx copy
import { useModal } from '@norr/components';
import { Portal } from './Portal';

const Component = () => {
  const {
    modalProps,
    modalTriggerProps,
    modalCloseProps,
    modalIsOpen,
  } = useModal<HTMLDivElement>();

  return (
    <>
      <button {...modalTriggerProps}>Trigger modal</button>
      {modalIsOpen && (
        <Portal>
          <div style={{ position: 'fixed', zIndex: 1337, inset: 0 }}>
            <div {...modalProps} style={{ backgroundColor: 'teal' }}>
              <span>Modal content.</span>
              <button {...modalCloseProps}>Close modal</button>
            </div>
          </div>
        </Portal>
      )}
    </>
  );
};
```

### Click outside

The component is using [useClickOutside()](/hooks/use-click-outside) under the
hood, so by default will the modal be closed when the user clicks outside of
the modal (where you spread `{...modalProps}`). This can be disabled by setting
the `disableClickOutside` prop to `true`.

<DemoProvider>
  <DemoModal disableClickOutside />
</DemoProvider>

```tsx copy {11,21}
import { useModal } from '@norr/components';
import { Portal } from './Portal';

const Component = () => {
  const {
    modalProps,
    modalTriggerProps,
    modalCloseProps,
    modalIsOpen,
  } = useModal<HTMLDivElement>({
    disableClickOutside: true,
  });

  return (
    <>
      <button {...modalTriggerProps}>Trigger modal</button>
      {modalIsOpen && (
        <Portal>
          <div style={{ position: 'fixed', zIndex: 1337, inset: 0 }}>
            <div {...modalProps} style={{ backgroundColor: 'teal' }}>
              <span>Will not be closed when clicked outside.</span>
              <button {...modalCloseProps}>Close modal</button>
            </div>
          </div>
        </Portal>
      )}
    </>
  );
};
```

### Escape key

The component is using [useEventListener()](/hooks/use-event-listener) under the
hood to listen for when the escape key is pressed, so by default will the modal
be closed when the user presses the `escape` key. This can be disabled by setting
the `disableEscapeKey` prop to `true`.

</DemoProvider>

### Prevent scroll

### Trap focus

The component is using [useTrapFocus()](/hooks/use-trap-focus) under the hood,
which will trap the focus within the modal. This can be disabled by setting
the `disableTrapFocus` prop to `true`.

<DemoProvider>
  <DemoModal disableTrapFocus />
</DemoProvider>

```tsx copy {11,21}
import { useModal } from '@norr/components';
import { Portal } from './Portal';

const Component = () => {
  const {
    modalProps,
    modalTriggerProps,
    modalCloseProps,
    modalIsOpen,
  } = useModal<HTMLDivElement>({
    disableTrapFocus: true,
  });

  return (
    <>
      <button {...modalTriggerProps}>Trigger modal</button>
      {modalIsOpen && (
        <Portal>
          <div style={{ position: 'fixed', zIndex: 1337, inset: 0 }}>
            <div {...modalProps} style={{ backgroundColor: 'teal' }}>
              <span>The focus is not trapped within the modal.</span>
              <button {...modalCloseProps}>Close modal</button>
            </div>
          </div>
        </Portal>
      )}
    </>
  );
};
```

### Return focus

The component is using [useTrapFocus()](/hooks/use-trap-focus) under the hood,
which will return the focus to the element that was focused before the modal was
mounted. This can be disabled by setting the `disableReturnFocus` prop to `true`.

<DemoProvider>
  <DemoModal disableReturnFocus />
</DemoProvider>

```tsx copy {11,22-23}
import { useModal } from '@norr/components';
import { Portal } from './Portal';

const Component = () => {
  const {
    modalProps,
    modalTriggerProps,
    modalCloseProps,
    modalIsOpen,
  } = useModal<HTMLDivElement>({
    disableReturnFocus: true,
  });

  return (
    <>
      <button {...modalTriggerProps}>Trigger modal</button>
      {modalIsOpen && (
        <Portal>
          <div style={{ position: 'fixed', zIndex: 1337, inset: 0 }}>
            <div {...modalProps} style={{ backgroundColor: 'teal' }}>

                The focus will not be returned to the element that was focused
                before the modal was mounted.
              </span>
              <button {...modalCloseProps}>Close modal</button>
            </div>
          </div>
        </Portal>
      )}
    </>
  );
};
```
