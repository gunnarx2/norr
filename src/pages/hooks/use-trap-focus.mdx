# useTrapFocus()

Trap keyboard focus inside a DOM element.

## Usage

When the component is mounted, the focus is trapped inside the element.

```tsx copy
import { useTrapFocus } from '@norr/hooks';

const Component = () => {
  const trapFocusRef = useTrapFocus<HTMLDivElement>();

  return (
    <div ref={trapFocusRef}>
      The focus is now trapped within this element.
      <form>
        <input type="text" />
        <button type="submit">Submit</button>
      </form>
    </div>
  );
};
```

### Return focus

When the element is unmounted, the focus is returned to the element that had
focus before the trap was activated. You can disable this by passing the
`disableReturnFocus` prop.

```tsx copy {5,10}
import { useTrapFocus } from '@norr/hooks';

const Component = () => {
  const trapFocusRef = useTrapFocus<HTMLDivElement>({
    disableReturnFocus: true,
  });

  return (
    <div ref={trapFocusRef}>
      The focus is not returned when this component unmounts.
      <form>
        <input type="text" />
        <button type="submit">Submit</button>
      </form>
    </div>
  );
};
```

### Disable

You can disable the trap by passing the `isDisabled` prop.  

```tsx copy {5,10}
import { useTrapFocus } from '@norr/hooks';

const Component = () => {
  const trapFocusRef = useTrapFocus<HTMLDivElement>({
    isDisabled: true,
  });

  return (
    <div ref={trapFocusRef}>
      The focus is not trapped within this element.
      <form>
        <input type="text" />
        <button type="submit">Submit</button>
      </form>
    </div>
  );
};
```
